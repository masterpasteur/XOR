\subsection{Evangélisation}

	\subsubsection{Réactions des collègues}
	
	\dida{Les collègues entrent en scène, l'un à droite et l'autre
		à gauche d'Henry.}
	\begin{itemize}
	\repItem{\henry}{Et c'est comme ça que je suis arrivé ici.}
	\repItem{\collIn}{OK, d'accord. Je comprends mieux.}
	\repItem{\collOut}{Mais j'ai pas compris pourquoi t'as autant
		hésité.}
	\repItem{\henry}{C'est parce que, forcément, un choix excluait l'autre.
		Et je ne voulais pas me tromper.}
	\repItem{\collOut}{M'ouais.}
	\repItem{\collIn}{Aller aider les pauvres petits Africains,
		ce n'était pas une erreur. C'est même bien !}
	\repItem{\henry}{C'est vrai.}
	\repItem{\collIn}{Alors, pourquoi t'es resté ?}
	\repItem{\henry}{En fait, j'ai toujours voulu être missionnaire.
		Oui, partir en mission en Afrique pour Dieu.}
	\repItem{\collOut}{Mais\ldots}
	\repItem{\henry}{Mais je me suis rendu compte que je pouvais servir
		Dieu tout en restant ici, à Paris.}
	\repItem{\collOut}{Depuis tout à l'heure, tu parles de Dieu. 
		Mais c'est vague.}
	\repItem{\collIn}{Ouais, en quoi tu crois ?}
	\end{itemize}
	
	\subsubsection{L'Evangile simplement}
	
	\dida{Henry ne sait pas par où commencer.}
	\begin{itemize}
	\repItem{\henry}{Hum\ldots}
	\repItem{\collIn}{On en reparle une prochaine fois, si tu veux.}
	\repItem{\henry}{Non, non. Je peux en parler maintenant.}
	\repItem{\collIn}{OK.}
	\repItem{\henry}{
		\ofg{Pourquoi cette injustice ? Pourquoi on m'a nourri, logé,
		éduqué alors que d'autres n'ont pas eu cette chance ?
		Pourquoi ?} J'étais incapable de répondre à ces questions.
		Je cherchais des réponses. J'ai commencé à lire la Bible.
		Et j'ai même mis les pieds dans une église.
	}
	\repItem{\collOut}{T'étais vraiment désespéré.}
	\repItem{\collIn}{Et t'avais pas peur ? C'était peut-être dangereux.}
	\repItem{\henry}{
		Personne n'est parfait, y compris dans les églises.
		Mais des personnes ont bien voulu répondre à mes questions.
		Et je n'avais pas peur parce que j'étais prêt à me remettre
		en question.
	}
	\repItem{\collOut}{C'est-à-dire ?}
	\repItem{\henry}{
		\ofg{Est-ce que Dieu pourrait exister ?
		Est-ce que je peux croire dans ce qui est écrit dans 
		la Bible ?}
		Après tout, avec nos horoscopes et nos délires sur le
		21 décembre, on croit déjà en n'importe quoi.
	}
	\repItem{\collOut}{\dida{Elle s'emporte.} 
		Ne critique pas les horoscopes, s'il te plaît !
		Et pour hier, c'était juste une erreur de calculs.}
	\repItem{\henry}{Désolé.}
	\repItem{\collIn}{Continue, Henry.}
	\repItem{\henry}{Les personnes de cette église m'ont expliqué ce
		que signifiait Noël.
		\ofg{On ne fête pas la venue d'un bébé. Mais celle d'un Roi.
		Le but, ce n'est pas de s'offrir des cadeaux que l'on revendra
		le lendemain. Mais c'est d'en recevoir un que l'on gardera pour
		toujours.}
	}
	\repItem{\collOut}{C'est trop facile de critiquer. Mais c'est quoi
		ce \ofg{cadeau} ?}
	\repItem{\henry}{
		\ofg{C'est la vie d'un homme qui s'offre en sacrifice.
		C'est la mort d'un héros pour sauver les hommes.}
	}
	\repItem{\collIn}{Elle est triste ton histoire. Et c'est pour nous
		sauver de quoi ?}
	\repItem{\henry}{De tout ce qui ne va pas dans ta vie.
		Tout ce qui t'attriste. De la mort, par exemple.
		Mais Jésus était innocent. C'est pour ça qu'il est revenu à la vie.
	}
	\repItem{\collOut}{Tu délires totalement.}
	\repItem{\henry}{Peut-être. Mais j'assume.}
	\repItem{\collIn}{C'est trop beau pour être vrai. On dirait un film.}
	\repItem{\henry}{Peut-être que les films se sont inspirés d'une 
		\emph{\oe uvre originale}.}
	\repItem{\collOut}{\dida{Elle regarde son téléphone}.
		Bon, la pause est terminée. On retourne bosser.}
	\repItem{\collIn}{C'était intéressant. On en reparle une prochaine
		fois?}
	\repItem{\henry}{Aucun problème.}
	\end{itemize}
	\dida{Ils sortent de la scène.}
